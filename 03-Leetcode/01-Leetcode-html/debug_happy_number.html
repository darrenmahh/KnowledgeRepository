🐛 快乐数代码调试：错误分析与修正

🔍 发现的问题：

❌ 问题1： 没有更新n的值，导致第二轮循环时n=0
❌ 问题2： sum重置时机错误，应该在更新n之后重置
❌ 问题3： 外层循环条件逻辑混乱，应该检查n而不是sum
❌ 你的错误代码
class Solution { public boolean isHappy(int n) { Set<Integer> set1 = new HashSet<>(); int sum = 0; int digit = 0; set1.add(n); while(sum != 1){ // 问题3: sum一开始是0 while(n > 0) { digit = n % 10; sum = sum + digit * digit; n = n / 10; } // 问题1: 没有 n = sum; if(set1.contains(sum)) return false; set1.add(sum); sum = 0; // 问题2: 重置太早 } return true; } }
✅ 修正后的代码
class Solution { public boolean isHappy(int n) { Set<Integer> seen = new HashSet<>(); while (n != 1 && !seen.contains(n)) { seen.add(n); int sum = 0; // sum声明在循环内 while (n > 0) { int digit = n % 10; sum += digit * digit; n /= 10; } n = sum; // 关键：更新n为新的值 } return n == 1; } }

class Solution { public boolean isHappy(int n) { Set<Integer> seen = new HashSet<>(); while (n != 1) { if (seen.contains(n)) { return false; // 检测到循环 } seen.add(n); n = getNext(n); // 提取为独立方法 } return true; } private int getNext(int n) { int sum = 0; while (n > 0) { int digit = n % 10; sum += digit * digit; n /= 10; } return sum; } }
